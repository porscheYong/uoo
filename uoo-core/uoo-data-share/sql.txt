drop table RABBITMQ_INDEX cascade constraints;
drop table NODE_INFO cascade constraints;
drop table RABBITMQ_USER_INFO cascade constraints;
drop table SYSTEM_QUEUE_RELA cascade constraints;

create table RABBITMQ_INDEX 
(
   id                 NUMBER(20)           not null,
   rabbitmq_date      CLOB,
   state              VARCHAR2(2),
   collection_data    DATE,
   use_data           DATE,
   constraint PK_RABBITMQ_INDEX primary key (id)
);

create table NODE_INFO 
(
   ID                   NUMBER(10)           not null,
   IP                   VARCHAR2(20),
   PORT                 VARCHAR2(10),
   constraint PK_NODE_INFO primary key (ID)
);

create table RABBITMQ_USER_INFO 
(
   ID                   NUMBER(10)           not null,
   USERNAME             VARCHAR2(40),
   PASSWORD             VARCHAR2(40),
   constraint PK_RABBITMQ_USER_INFO primary key (ID)
);

create table SYSTEM_QUEUE_RELA 
(
   QUEUE_NAME           VARCHAR2(50)         not null,
   SYSTEM_NAME          VARCHAR2(50),
   DOUBLE_NAME          VARCHAR2(50),
   CHARGE_PERSON        VARCHAR2(20),
   CHARGE_CONTACT       VARCHAR2(20),
   STATUS               VARCHAR2(4),
   IP                   VARCHAR2(20),
   PORT                 VARCHAR2(10),
   USERNAME             VARCHAR2(40),
   PASSWORD             VARCHAR2(40),
   constraint PK_SYSTEM_QUEUE_RELA primary key (QUEUE_NAME)
);

create sequence SEQ_TB_RABBITMQ_INDEX_ID
